# add acolumn for age_classifier
def age_classifier(age):
    if age >= 0 and age <18:
        return "Kids"
    elif age >= 18 and age < 37:
        return "Adults"
    elif age >= 37 and age < 55:
        return "Matures"
    else:
        return "Elders"

ford_clean["age_group"] = ford_clean["age"].apply(age_classifier)
